
## contextual  

### warning 

```jsx live=true
<Alert kind="warning">
    A warning alert
</Alert>
```

### danger with custom icon

```jsx live=true
<Alert icon={props=> <GitBranch {...props} size={25}/>} kind="danger">
    Danger will robinson with custom icon
</Alert>
```

### error

```jsx live=true
<Alert kind="error">
    An error alert
</Alert>
```

### success 

```jsx live=true
<Alert kind="success" dismissable>
    A successful alert
</Alert>
```
